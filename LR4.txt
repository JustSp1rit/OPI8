#include <iostream>
#include "stdio.h"
#include "math.h" 

using namespace std;
int main()
{
double a, x, y;
printf("Введіть x: \n");
scanf("%lf", &x);
printf("Введіть y: \n");
scanf("%lf", &y);
a = sqrt(1.0/(2.0*pow(x,-3.0))) + 
  pow((pow(y,3.0) + 3.0 * pow(y,2.0) - 18.0*y) / (abs(x-y)),1/3.0) + exp(x+y);



printf("a=%lf", a);

return 0;
}
